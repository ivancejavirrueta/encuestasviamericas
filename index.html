    <script>

      async function enviar(valor) {

        try {

          // Reemplaza ESTA URL con la del despliegue Web App de tu Google Apps Script

          const scriptURL = "https://script.google.com/macros/s/TU_DEPLOY_ID/exec";

          const resp = await fetch(scriptURL, {

            method: "POST",

            headers: { "Content-Type": "application/x-www-form-urlencoded" },

            body: `respuesta=${encodeURIComponent(valor)}`

          });

          if (resp.ok) {

            document.getElementById('msg').innerText = "¡Gracias por tu opinión!";

          } else {

            document.getElementById('msg').innerText = "Error al enviar respuesta.";

          }

        } catch (err) {

          console.error(err);

          document.getElementById('msg').innerText = "Error de conexión.";

        }

      }

    </script>

    <script>

      if ('serviceWorker' in navigator) {

        navigator.serviceWorker.register('sw.js')

          .then(() => console.log('SW registrado'))

          .catch(err => console.error('Error SW:', err));

      }

    </script>
